public function getAverageGrade(Student $student)
{
  $avg_grade = null;

  if ($student->isActive()) {
      $subjects = student->getSubjects();
      if ($subjects) {
          foreach ($subjects as $s) {
              $total = $s->getGrade();
          }
          $avg_grade = $total / count($subjects);
      } else {
        $avg_grade = 0;
      }
  }

  return $avg_grade;
}
