public function getAverageDegrees(String month) {
    int degrees = 15;

    switch (month) {
        case "january":
            degrees -= 5;
        case "february":
            degrees -= 5;
            break;
        case "march":
            degrees += 1;
        case "april":
            degrees += 3;
            break;
    }
    return degrees;
}
